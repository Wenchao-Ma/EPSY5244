tab <- table(assoc_df$housetype, assoc_df$pref)
tab

### Chi-Square Test and Cramer's V

```{r q2-chi-square}
chi_fit <- chisq.test(tab, correct = FALSE)
chi_fit

# Cramer's V
n <- sum(tab)
r <- nrow(tab)
c <- ncol(tab)
cramers_v <- sqrt(as.numeric(chi_fit$statistic) / (n * min(r - 1, c - 1)))
cramers_v
```

- Cramer's $V$: `r round(cramers_v, 3)`
- Row percentages:
  - Senior Housing preferring Relocate: `r round(100 * prop.table(tab, 1)["Senior Housing", "Relocate"], 1)`%
  - Family Housing preferring Relocate: `r round(100 * prop.table(tab, 1)["Family Housing", "Relocate"], 1)`%
- Interpretation: housing type is significantly associated with relocation preference. Family Housing residents are more likely to prefer relocation, while Senior Housing residents are more likely to prefer fixing up current housing.

## Q5: Logistic Regression

**Question:** Which factors predict preference for relocation (vs. fixing up current housing)?

Outcome:

- `Q134W1` recoded as 1 = Relocate, 0 = Fix Up

Predictors:

- `Houstype` (0 = Senior Housing, 1 = Family Housing)
- `Q137W1` (confidence about post-relocation housing quality; higher values = less confidence)
- `Age`

```{r q5-logit-fit}
logit_df <- df %>%
  transmute(
    relocate = ifelse(as.numeric(Q134W1) == 2, 1,
                      ifelse(as.numeric(Q134W1) == 1, 0, NA)),
    housetype = as.numeric(Houstype),
    confidence = as.numeric(Q137W1),
    age = as.numeric(Age)
  ) %>%
  filter(
    !is.na(relocate),
    housetype %in% c(0, 1),
    confidence %in% 1:4,
    !is.na(age)
  )

logit_fit <- glm(
  relocate ~ housetype + confidence + age,
  data = logit_df,
  family = binomial(link = "logit")
)

summary(logit_fit)
```

```{r q5-logit-or-table}
or_table <- data.frame(
  term = names(coef(logit_fit)),
  OR = exp(coef(logit_fit)),
  conf.low = exp(confint(logit_fit)[, 1]),
  conf.high = exp(confint(logit_fit)[, 2]),
  p.value = summary(logit_fit)$coefficients[, 4],
  row.names = NULL
)

or_table_display <- or_table %>%
  mutate(across(where(is.numeric), ~ round(.x, 3)))

knitr::kable(or_table_display, caption = "Odds ratios for relocation preference")
```

::: {.note-box .exercise}
### Answer (from this dataset)

- Significant predictors at $\alpha=.05$ are those with `p.value < .05` in the odds-ratio table.
- Interpretations:
  - `Houstype` OR $> 1$ indicates Family Housing residents have higher odds of preferring relocation than Senior Housing residents.
  - `confidence` OR $> 1$ indicates that lower confidence in post-relocation housing quality is associated with higher odds of preferring relocation.
  - `age` OR indicates how odds of preferring relocation change per one-year increase in age.
:::

## Q6 (Optional): Missing Pattern and Multiple Imputation

**Question:** Assess the missing-data pattern and perform multiple imputation (MI) before analysis. Compare MI-based results with complete-case results.

```{r q6-missing-summary}
mi_vars <- df %>%
  transmute(
    Q134W1 = as.numeric(Q134W1),
    Houstype = as.numeric(Houstype),
    Q132W1 = as.numeric(Q132W1),
    Q137W1 = as.numeric(Q137W1),
    Q139aW1 = as.numeric(Q139aW1),
    Q139bW1 = as.numeric(Q139bW1),
    Q139cW1 = as.numeric(Q139cW1),
    Q139dW1 = as.numeric(Q139dW1),
    Age = as.numeric(Age)
  )

miss_pct <- sapply(mi_vars, function(x) mean(is.na(x)) * 100)
overall_any_missing_pct <- mean(!complete.cases(mi_vars)) * 100
max_miss_var <- names(which.max(miss_pct))
max_miss_pct <- max(miss_pct)
```

```{r q6-compare-q2}
# Q2 comparison: complete-case vs MI
assoc_dat <- mi_vars %>%
  transmute(
    pref = ifelse(Q134W1 %in% c(1, 2), Q134W1, NA_real_),
    housetype = ifelse(Houstype %in% c(0, 1), Houstype, NA_real_)
  )

assoc_cc <- assoc_dat %>% filter(complete.cases(.))
tab_cc <- table(assoc_cc$housetype, assoc_cc$pref)
chi_cc <- chisq.test(tab_cc, correct = FALSE)
v_cc <- sqrt(as.numeric(chi_cc$statistic) / (sum(tab_cc) * min(nrow(tab_cc) - 1, ncol(tab_cc) - 1)))

imp_assoc <- mice(assoc_dat, m = 20, maxit = 10, seed = 5244, printFlag = FALSE)
assoc_list <- complete(imp_assoc, action = "all")

chi_stats <- sapply(assoc_list, function(dd) {
  tt <- table(dd$housetype, dd$pref)
  as.numeric(chisq.test(tt, correct = FALSE)$statistic)
})
p_vals <- sapply(assoc_list, function(dd) {
  tt <- table(dd$housetype, dd$pref)
  chisq.test(tt, correct = FALSE)$p.value
})
v_vals <- sapply(assoc_list, function(dd) {
  tt <- table(dd$housetype, dd$pref)
  sqrt(as.numeric(chisq.test(tt, correct = FALSE)$statistic) /
         (sum(tt) * min(nrow(tt) - 1, ncol(tt) - 1)))
})

chi_mi_med <- median(chi_stats)
p_mi_med <- median(p_vals)
v_mi_med <- median(v_vals)
```

```{r q6-compare-q5}
# Q5 comparison: complete-case vs MI
q5_dat <- df %>%
  transmute(
    relocate = ifelse(as.numeric(Q134W1) == 2, 1,
                      ifelse(as.numeric(Q134W1) == 1, 0, NA)),
    housetype = ifelse(as.numeric(Houstype) %in% c(0, 1), as.numeric(Houstype), NA_real_),
    confidence = ifelse(as.numeric(Q137W1) %in% 1:4, as.numeric(Q137W1), NA_real_),
    age = as.numeric(Age)
  )

q5_cc <- q5_dat %>% filter(complete.cases(.))
fit_q5_cc <- glm(relocate ~ housetype + confidence + age, data = q5_cc, family = binomial())
cc_coef <- coef(summary(fit_q5_cc))

imp_q5 <- mice(q5_dat, m = 20, maxit = 10, seed = 5244, printFlag = FALSE)
fit_q5_mi <- with(imp_q5, glm(relocate ~ housetype + confidence + age, family = binomial()))
pool_q5_mi <- summary(pool(fit_q5_mi))

cc_houstype_p <- cc_coef["housetype", "Pr(>|z|)"]
cc_confidence_p <- cc_coef["confidence", "Pr(>|z|)"]
cc_age_p <- cc_coef["age", "Pr(>|z|)"]

mi_houstype_p <- pool_q5_mi$p.value[pool_q5_mi$term == "housetype"]
mi_confidence_p <- pool_q5_mi$p.value[pool_q5_mi$term == "confidence"]
mi_age_p <- pool_q5_mi$p.value[pool_q5_mi$term == "age"]
```

  - MI pooled p-values: Houstype `r format.pval(mi_houstype_p, digits = 4)`, confidence `r format.pval(mi_confidence_p, digits = 4)`, age `r format.pval(mi_age_p, digits = 4)`.
- Interpretation: MI does not materially change substantive conclusions in this assignment, so complete-case analysis and MI lead to similar decisions.
